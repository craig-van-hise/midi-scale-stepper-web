### 1. Intercept the Stepper Zone

* **ELSE:** Continue existing pass-through logic.

### 2. Implement `calculateStep(incomingNote)`
This is the "Brain" of the MVP. Write this function to perform the scalar index calculation:

```typescript
function calculateStep(incomingNote: number) {
  // 1. Get current state from Zustand
  const { activeState, activeState: { lastPlayedMidi, scaleDecimalId } } = useMidiStore.getState();
  const lut = getLUT();
  const entry = lut[scaleDecimalId!];
  if (!entry) return;

  // 2. Define the step interval (C5 is index 0)
  const stepOffset = incomingNote - 72; 

  // 3. Determine current scale index
  // Use the lastPlayedMidi to find the current scale degree
  const currentPC = (lastPlayedMidi ?? (60 + (activeState.rootNote ?? 0))) % 12;
  const currentIndex = entry.pitch_classes.indexOf(currentPC);
  
  // 4. Calculate new index
  // Clamp the new index within the bounds of the current scale array
  let newIndex = currentIndex + stepOffset;
  const scaleLength = entry.pitch_classes.length;
  
  // Handle wrapping (Octave shift)
  let octaveShift = Math.floor(newIndex / scaleLength);
  let wrappedIndex = ((newIndex % scaleLength) + scaleLength) % scaleLength;
  
  // 5. Get the new Pitch Class and reconstruct MIDI note
  const newPC = entry.pitch_classes[wrappedIndex];
  const currentOctave = Math.floor((lastPlayedMidi ?? 72) / 12);
  const outputMidi = ((currentOctave + octaveShift) * 12) + (activeState.rootNote ?? 0) + newPC;

  // 6. Update State & Route
  useMidiStore.getState().setActiveState({ ...activeState, lastPlayedMidi: outputMidi });
  sendMidiToOutput(outputMidi); // Trigger the note
}

```
